# Repora Backend (Full Hermes Spec) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Flask + Python backend implementing the full Hermes multi-agent document-orchestration platform from `AGENTS.md`, containerized alongside the existing React frontend, plus professional READMEs for both.

**Architecture:** 3-tier + agent layer. Flask REST + SSE API over PostgreSQL (SQLAlchemy 2.0 async + Alembic). A `Hermes` orchestrator dispatches registered agents through a shared OpenAI-compatible tool-calling loop, backed by a provider-abstraction layer (local `llama-server` by default, BYOK opt-in). Each of `backend/` and `repora-web/` has its own Dockerfile; root `docker-compose.yml` wires frontend + backend + postgres + llama-server.

**Tech Stack:** Python 3.12, Flask 3, Flask-CORS, SQLAlchemy 2.0 (async), Alembic, pydantic v2 + pydantic-settings, Authlib/JWT (PyJWT), passlib[argon2], httpx (provider clients + SSE), PlantUML (server or `plantuml` lib), python-docx + reportlab (export), pytest, gunicorn/uvicorn.

## Global Constraints

- Backend language: **Flask + Python 3.12** (per AGENTS.md), monorepo sibling `backend/` next to `repora-web/`.
- Each of `backend/` and `repora-web/` ships its **own Dockerfile**; root `docker-compose.yml` orchestrates `frontend`, `backend`, `db` (postgres:17), `llama-server`.
- DB layer: **SQLAlchemy 2.0 async + Alembic**.
- LLM: **`llama-server` real HTTP client** (OpenAI-compatible `/v1/chat/completions`, `stream:true`) as default; BYOK providers opt-in. Clear "model unavailable" error if `llama-server` is down — never crash.
- Provider interface is provider-agnostic: both implement `ModelProvider.stream_chat(messages, tools, cfg) -> async iterator[dict]`.
- All timestamps stored as ISO-8601 UTC strings; all JSON responses wrapped in a consistent envelope `{ "data": ... }` or `{ "error": { "code", "message" } }`.
- TDD: each task writes a failing test first, then minimal implementation, then runs it green, then commits.
- Secrets via env (pydantic-settings); never log API keys or tokens.
- RBAC roles: `owner`, `admin`, `editor`, `reviewer`, `viewer` (mirror `schemas/index.ts`).

```python
# backend/app/db/session.py
from contextlib import asynccontextmanager
from app.db.base import async_session_factory, AsyncSession


@asynccontextmanager
async def get_session() -> AsyncSession:
    session = async_session_factory()
    try:
        yield session
        await session.commit()
    except Exception:
        await session.rollback()
        raise
    finally:
        await session.close()
```

```python
# backend/migrations/env.py
import asyncio
from logging.config import fileConfig
from alembic import context
from sqlalchemy.ext.asyncio import async_engine_from_config
from sqlalchemy import pool
from app.db.base import Base
from app.db import models  # ensure models imported
from app.config import settings

config = context.config
config.set_main_option("sqlalchemy.url", settings.database_url)
if config.config_file_name:
    fileConfig(config.config_file_name)
target_metadata = Base.metadata


def run_migrations_offline():
    context.configure(url=settings.database_url, target_metadata=target_metadata, literal_binds=True)
    with context.begin_transaction():
        context.run_migrations()


def do_run_migrations(connection):
    context.configure(connection=connection, target_metadata=target_metadata)
    with context.begin_transaction():
        context.run_migrations()


async def run_migrations_online():
    connectable = async_engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)
    await connectable.dispose()


if context.is_offline_mode():
    run_migrations_offline()
else:
    asyncio.run(run_migrations_online())
```

```python
from functools import wraps
from flask import request, jsonify, g
from app.auth.jwt import decode_token


def _extract():
    h = request.headers.get("Authorization", "")
    if not h.startswith("Bearer "):
        return None
    try:
        return decode_token(h[7:])
    except Exception:
        return None


def login_required(f):
    @wraps(f)
    def wrapper(*a, **k):
        data = _extract()
        if not data:
            return jsonify(error={"code": "unauthorized", "message": "Login required"}), 401
        g.user = data
        return f(*a, **k)
    return wrapper


def role_required(*roles):
    def decorator(f):
        @wraps(f)
        def wrapper(*a, **k):
            data = _extract()
            if not data:
                return jsonify(error={"code": "unauthorized", "message": "Login required"}), 401
            if roles and data.get("role") not in roles:
                return jsonify(error={"code": "forbidden", "message": "Insufficient role"}), 403
            g.user = data
            return f(*a, **k)
        return wrapper
    return decorator
```

```python
# backend/app/agents/loop.py
import json
from app.agents.tools import call_tool


def run_tool_loop(provider, messages: list[dict], tools: list[dict], cfg: dict) -> str:
    # Synchronous driver used by Hermes; provider.stream_chat yields SSE chunks.
    # We accumulate the assistant message, executing any tool_calls via call_tool.
    assistant = {"role": "assistant", "content": "", "tool_calls": []}
    for chunk in _collect(provider, messages, tools, cfg):
        delta = chunk.get("choices", [{}])[0].get("delta", {})
        if "content" in delta:
            assistant["content"] += delta["content"] or ""
        if "tool_calls" in delta:
            for tc in delta["tool_calls"]:
                _merge_tool_call(assistant, tc)
    # execute tool calls
    if assistant.get("tool_calls"):
        messages.append(assistant)
        for tc in assistant["tool_calls"]:
            args = json.loads(tc.get("function", {}).get("arguments", "{}") or "{}")
            result = call_tool(tc["function"]["name"], args)
            messages.append({"role": "tool", "tool_call_id": tc["id"], "content": json.dumps(result)})
        # one more pass for final content
        return run_tool_loop(provider, messages, tools, cfg)
    return assistant["content"]


def _collect(provider, messages, tools, cfg):
    import asyncio
    out = []
    async def _g():
        async for c in provider.stream_chat(messages, tools, cfg):
            out.append(c)
    asyncio.run(_g())
    return out


def _merge_tool_call(assistant, tc):
    idx = tc.get("index", 0)
    while len(assistant["tool_calls"]) <= idx:
        assistant["tool_calls"].append({"id": "", "function": {"name": "", "arguments": ""}})
    t = assistant["tool_calls"][idx]
    if tc.get("id"):
        t["id"] = tc["id"]
    if tc.get("function", {}).get("name"):
        t["function"]["name"] = tc["function"]["name"]
    if tc.get("function", {}).get("arguments"):
        t["function"]["arguments"] += tc["function"]["arguments"]
```

```python
# backend/app/agents/hermes.py
import asyncio
from app.agents.registry import get_registry
from app.agents.loop import run_tool_loop
from app.agents.providers.base import resolve_provider


class Hermes:
    def __init__(self, project_id: str, document_id: str):
        self.project_id = project_id
        self.document_id = document_id

    def run(self, prompt: str) -> dict:
        registry = {a["name"]: a for a in get_registry()}
        # Planner -> Writer -> Reviewer (sequential graph)
        outline = self._step(registry["Planner"], prompt)
        sections = self._step(registry["Writer"], f"Outline: {outline}\nWrite sections for document {self.document_id}.")
        review = self._step(registry["Reviewer"], f"Review document {self.document_id}.")
        return {"outline": outline, "sections_written": sections, "review": review}

    def _step(self, agent: dict, prompt: str) -> str:
        provider = resolve_provider(agent["provider"])
        messages = [
            {"role": "system", "content": agent["system_prompt"]},
            {"role": "user", "content": prompt},
        ]
        tools = [{"type": "function", "function": {"name": t, "parameters": {"type": "object", "properties": {}}}} for t in agent["tools"]]
        cfg = {"model": agent.get("model", "local"), "temperature": 0.7, "top_p": 0.9, "max_tokens": 1024}
        return run_tool_loop(provider, messages, tools, cfg)
```

```python
# backend/app/streaming/hub.py
import asyncio, queue


class Hub:
    def __init__(self):
        self._subs = {}

    def subscribe(self, key):
        q = asyncio.Queue()
        self._subs.setdefault(key, []).append(q)
        return q

    async def publish(self, key, event: dict):
        for q in self._subs.get(key, []):
            await q.put(event)


hub = Hub()
```

```python
# backend/app/services/plantuml.py
def render(plantuml_source: str, server_url: str = "http://localhost:8080") -> str:
    # Returns a URL to the rendered diagram (uses PlantUML server proxy endpoint).
    import urllib.parse, base64, zlib
    data = zlib.compress(plantuml_source.encode("utf-8"), 9)
    encoded = base64.b64encode(data).decode("ascii")
    return f"{server_url}/svg/{encoded}"
```

```python
# backend/app/services/export.py
from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from reportlab.lib.styles import getSampleStyleSheet
from io import BytesIO


def export_pdf(title: str, sections: list[dict]) -> bytes:
    buf = BytesIO()
    doc = SimpleDocTemplate(buf, pagesize=letter)
    styles = getSampleStyleSheet()
    story = [Paragraph(title, styles["Title"])]
    for s in sections:
        story.append(Paragraph(s.get("title", ""), styles["Heading2"]))
        story.append(Paragraph(s.get("content", ""), styles["BodyText"]))
        story.append(Spacer(1, 12))
    doc.build(story)
    return buf.getvalue()


def export_docx(title: str, sections: list[dict]) -> bytes:
    from docx import Document as DocxDocument
    d = DocxDocument()
    d.add_heading(title, level=0)
    for s in sections:
        d.add_heading(s.get("title", ""), level=2)
        d.add_paragraph(s.get("content", ""))
    buf = BytesIO()
    d.save(buf)
    return buf.getvalue()
```

```python
@role_required("owner", "admin")
def patch_agent(name):
    body = request.get_json(force=True) or {}
    async def _do():
        async with get_session() as s:
            r = (await s.execute(sqlalchemy.select(AgentConfig).where(AgentConfig.agent_name == name))).scalar_one_or_none()
```
